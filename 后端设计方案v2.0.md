#### 2.2.2 OKR 变更历史触发器
```sql
CREATE OR REPLACE FUNCTION log_okr_changes()
RETURNS TRIGGER AS $$
BEGIN
    IF TG_OP = 'INSERT' THEN
        INSERT INTO okr_history (okr_id, changed_by, action, new_value)
        VALUES (NEW.id, NEW.created_by, 'CREATE', row_to_json(NEW)::text);
        RETURN NEW;
    ELSIF TG_OP = 'UPDATE' THEN
        -- 记录状态变更
        IF OLD.status != NEW.status THEN
            INSERT INTO okr_history (okr_id, changed_by, action, field_name, old_value, new_value)
            VALUES (NEW.id, NEW.updated_by, 'STATUS_CHANGE', 'status', OLD.status, NEW.status);
        END IF;
        -- 记录评分变更
        IF OLD.total_score != NEW.total_score THEN
            INSERT INTO okr_history (okr_id, changed_by, action, field_name, old_value, new_value)
            VALUES (NEW.id, NEW.updated_by, 'SCORE_UPDATE', 'total_score', 
                    OLD.total_score::text, NEW.total_score::text);
        END IF;
        RETURN NEW;
    END IF;
    RETURN NULL;
END;
$$ language 'plpgsql';

CREATE TRIGGER okr_history_trigger
    AFTER INSERT OR UPDATE ON okrs
    FOR EACH ROW EXECUTE FUNCTION log_okr_changes();
```

```typescript
// src/utils/okrStateMachine.ts

export enum OKRStatus {
  DRAFT = 'DRAFT',
  PENDING_MANAGER = 'PENDING_MANAGER',
  PENDING_GM = 'PENDING_GM',
  PUBLISHED = 'PUBLISHED',
  PENDING_ASSESSMENT_APPROVAL = 'PENDING_ASSESSMENT_APPROVAL',
  GRADING = 'GRADING',
  PENDING_L2_APPROVAL = 'PENDING_L2_APPROVAL',
  PENDING_L3_APPROVAL = 'PENDING_L3_APPROVAL',
  PENDING_ARCHIVE = 'PENDING_ARCHIVE',
  CLOSED = 'CLOSED'
}

// 状态转换规则
export const STATUS_TRANSITIONS: Record<OKRStatus, OKRStatus[]> = {
  [OKRStatus.DRAFT]: [OKRStatus.PENDING_MANAGER],
  [OKRStatus.PENDING_MANAGER]: [OKRStatus.DRAFT, OKRStatus.PENDING_GM, OKRStatus.PUBLISHED],
  [OKRStatus.PENDING_GM]: [OKRStatus.DRAFT, OKRStatus.PUBLISHED],
  [OKRStatus.PUBLISHED]: [OKRStatus.PENDING_ASSESSMENT_APPROVAL],
  [OKRStatus.PENDING_ASSESSMENT_APPROVAL]: [
    OKRStatus.PENDING_L2_APPROVAL,
    OKRStatus.PENDING_ARCHIVE,
    OKRStatus.PENDING_ASSESSMENT_APPROVAL // 允许驳回
  ],
  [OKRStatus.GRADING]: [OKRStatus.PENDING_ASSESSMENT_APPROVAL],
  [OKRStatus.PENDING_L2_APPROVAL]: [
    OKRStatus.PENDING_L3_APPROVAL,
    OKRStatus.PENDING_ARCHIVE,
    OKRStatus.PENDING_ASSESSMENT_APPROVAL // 允许驳回
  ],
  [OKRStatus.PENDING_L3_APPROVAL]: [
    OKRStatus.PENDING_ARCHIVE,
    OKRStatus.PENDING_ASSESSMENT_APPROVAL // 允许驳回
  ],
  [OKRStatus.PENDING_ARCHIVE]: [OKRStatus.CLOSED],
  [OKRStatus.CLOSED]: [] // 终态，不允许转换
};

// 验证状态转换
export function canTransition(from: OKRStatus, to: OKRStatus): boolean {
  return STATUS_TRANSITIONS[from]?.includes(to) ?? false;
}

// 获取允许的状态转换
export function getAllowedTransitions(currentStatus: OKRStatus): OKRStatus[] {
  return STATUS_TRANSITIONS[currentStatus] ?? [];
}
```

### 3.2 状态转换前置条件

```typescript
// src/utils/okrValidators.ts

export interface StatusTransitionContext {
  okr: OKR;
  user: User;
  newStatus: OKRStatus;
  hasL2Approver: boolean;
  hasL3Approver: boolean;
}

export function validateStatusTransition(context: StatusTransitionContext): {
  valid: boolean;
  reason?: string;
} {
  const { okr, newStatus } = context;
  
  // 1. 检查状态转换是否允许
  if (!canTransition(okr.status, newStatus)) {
    return { valid: false, reason: `不允许从 ${okr.status} 转换到 ${newStatus}` };
  }
  
  // 2. 检查前置条件
  switch (newStatus) {
    case OKRStatus.PENDING_ASSESSMENT_APPROVAL:
      if (!okr.overallSelfAssessment?.comment) {
        return { valid: false, reason: '提交自评必须填写整体自评总结' };
      }
      break;
      
    case OKRStatus.PENDING_L2_APPROVAL:
      if (!okr.totalScore) {
        return { valid: false, reason: '一级评分必须完成' };
      }
      if (!context.hasL2Approver) {
        return { valid: false, reason: '没有配置二级审批人' };
      }
      break;
      
    case OKRStatus.PENDING_L3_APPROVAL:
      if (!context.hasL3Approver) {
        return { valid: false, reason: '没有配置三级审批人' };
      }
      break;
      
    case OKRStatus.PENDING_ARCHIVE:
      if (okr.status === OKRStatus.PENDING_L2_APPROVAL && !context.hasL3Approver) {
        // 允许：没有L3时，L2完成后直接归档
        break;
      }
      if (okr.status === OKRStatus.PENDING_L3_APPROVAL) {
        // L3完成后归档
        break;
      }
      if (!okr.totalScore) {
        return { valid: false, reason: '归档前必须完成评分' };
      }
      break;
  }
  
  return { valid: true };
}
```

### 4.1 审批权限验证

```typescript
// src/utils/permissionChecker.ts

export interface ApprovalPermission {
  canApprove: boolean;
  canReject: boolean;
  reason?: string;
}

export async function checkApprovalPermission(
  userId: string,
  okrId: string,
  action: 'approve' | 'reject'
): Promise<ApprovalPermission> {
  // 1. 获取 OKR 和用户信息
  const okr = await getOKRById(okrId);
  const user = await getUserById(userId);
  
  if (!okr || !user) {
    return { canApprove: false, canReject: false, reason: '数据不存在' };
  }
  
  // 2. 获取 OKR 创建者信息
  const okrOwner = await getUserById(okr.user_id);
  if (!okrOwner) {
    return { canApprove: false, canReject: false, reason: 'OKR 创建者不存在' };
  }
  
  // 3. 检查当前状态是否允许审批
  const allowedStatuses = [
    OKRStatus.PENDING_MANAGER,
    OKRStatus.PENDING_GM,
    OKRStatus.PENDING_ASSESSMENT_APPROVAL,
    OKRStatus.PENDING_L2_APPROVAL,
    OKRStatus.PENDING_L3_APPROVAL
  ];
  
  if (!allowedStatuses.includes(okr.status)) {
    return { canApprove: false, canReject: false, reason: '当前状态不允许审批' };
  }
  
  // 4. 根据状态和角色验证权限
  switch (okr.status) {
    case OKRStatus.PENDING_MANAGER:
      // 验证是否为 L1 审批人
      if (user.role !== okr.approver_l1_role) {
        return { canApprove: false, canReject: false, reason: '不是一级审批人' };
      }
      // 验证部门匹配（如果需要）
      if (okrOwner.department !== user.department && !isGlobalRole(user.role)) {
        return { canApprove: false, canReject: false, reason: '部门不匹配' };
      }
      break;
      
    case OKRStatus.PENDING_GM:
      if (user.role !== okr.approver_l2_role) {
        return { canApprove: false, canReject: false, reason: '不是二级审批人' };
      }
      break;
      
    case OKRStatus.PENDING_ASSESSMENT_APPROVAL:
      if (user.role !== okr.approver_l1_role) {
        return { canApprove: false, canReject: false, reason: '不是一级评分人' };
      }
      break;
      
    case OKRStatus.PENDING_L2_APPROVAL:
      if (user.role !== okr.approver_l2_role) {
        return { canApprove: false, canReject: false, reason: '不是二级审批人' };
      }
      break;
      
    case OKRStatus.PENDING_L3_APPROVAL:
      if (user.role !== okr.approver_l3_role) {
        return { canApprove: false, canReject: false, reason: '不是三级审批人' };
      }
      break;
  }
  
  // 5. 管理员特殊权限
  if (user.role === Role.ADMIN) {
    return { canApprove: true, canReject: true };
  }
  
  return { canApprove: true, canReject: true };
}

function isGlobalRole(role: string): boolean {
  const globalRoles = [
    Role.PRODUCT_GM,
    Role.TECH_GM,
    Role.VP_PRODUCT,
    Role.VP_TECH,
    Role.VP_MARKET,
    Role.PRESIDENT
  ];
  return globalRoles.includes(role as Role);
}
```

```typescript
export async function withTransaction<T>(
  callback: (client: PoolClient) => Promise<T>
): Promise<T> {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    const result = await callback(client);
    await client.query('COMMIT');
    return result;
  } catch (error) {
    await client.query('ROLLBACK');
    throw error;
  } finally {
    client.release();
  }
}
```

```typescript
      throw new Error('OKR not found');
    }
    
    const currentVersion = current.rows[0].version;
    
    // 2. 更新时检查版本
    const result = await client.query(
      `UPDATE okrs 
       SET ..., version = version + 1, updated_by = $1
       WHERE id = $2 AND version = $3
       RETURNING *`,
      [userId, id, currentVersion]
    );
    
    if (result.rows.length === 0) {
      throw new Error('版本冲突，请刷新后重试');
    }
    
    // 3. 记录变更历史
    // ...
  });
}
```

```typescript
import NodeCache from 'node-cache';

const cache = new NodeCache({ stdTTL: 300 }); // 5分钟过期

export async function getCachedWorkflows(): Promise<Workflow[]> {
  const cached = cache.get<Workflow[]>('workflows');
  if (cached) {
    return cached;
  }
  
  const workflows = await fetchWorkflowsFromDB();
  cache.set('workflows', workflows);
```
